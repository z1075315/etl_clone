CREATE OR REPLACE FUNCTION staging.fn_full_load_to_stg_banner()
 RETURNS integer
 LANGUAGE plpgsql
AS $function$
DECLARE
    v_step              INT;
    cur_csv_banner CURSOR FOR
    SELECT banner_id,
	       banner_name,
	       retailer_banner_id,
	       attr_1,
	       attr_2,
	       attr_3,
	       attr_4,
	       attr_5,
	       attr_6,
	       attr_7,
	       attr_8,
	       attr_9,
	       attr_10,
	       created_user, 
		   created_ts, 
		   updated_user, 
		   updated_ts
    FROM   staging.csv_banner;
BEGIN
    v_step := 1;
    FOR rec_cur_csv_banner IN cur_csv_banner LOOP
        BEGIN
            INSERT INTO staging.stg_banner 
			     (banner_id,
	              banner_name,
	              retailer_banner_id,
	              attr_1,
	              attr_2,
	              attr_3,
	              attr_4,
	              attr_5,
	              attr_6,
	              attr_7,
	              attr_8,
	              attr_9,
	              attr_10,
	              created_user, 
		          created_ts, 
		          updated_user, 
		          updated_ts)
            VALUES 
			     (rec_cur_csv_banner.banner_id,
	              rec_cur_csv_banner.banner_name,
	              rec_cur_csv_banner.retailer_banner_id,
	              rec_cur_csv_banner.attr_1,
	              rec_cur_csv_banner.attr_2,
	              rec_cur_csv_banner.attr_3,
	              rec_cur_csv_banner.attr_4,
	              rec_cur_csv_banner.attr_5,
	              rec_cur_csv_banner.attr_6,
	              rec_cur_csv_banner.attr_7,
	              rec_cur_csv_banner.attr_8,
	              rec_cur_csv_banner.attr_9,
	              rec_cur_csv_banner.attr_10,
				  rec_cur_csv_banner.created_user, 
		          rec_cur_csv_banner.created_ts, 
		          rec_cur_csv_banner.updated_user, 
		          rec_cur_csv_banner.updated_ts);
	    EXCEPTION 
            WHEN OTHERS THEN
                 INSERT INTO staging.error_logs(error_id, error_function_name, error_step_number, error_state, error_msg, primarykeyid) 
                 VALUES(nextval('staging.error_logs_pk_seq'), 'fn_full_load_to_stg_banner', v_step, sqlstate, sqlerrm, rec_cur_csv_banner.banner_id);
				 RETURN 1;
	    END;
    END LOOP;
    v_step := 2;
    RETURN 0;
END;
$function$
;
