CREATE OR REPLACE FUNCTION staging.fn_full_load_to_stg_merch_level1()
 RETURNS integer
 LANGUAGE plpgsql
AS $function$
DECLARE
    v_step              INT;
    cur_csv_merch_level1 CURSOR FOR
    SELECT merch_level1_id,
	       merch_level1_name,
	       merch_level1_desc,
	       active_f,
	       retailer_merch_level1_id,
	       attr_1,
	       attr_2,
	       attr_3,
	       attr_4,
	       attr_5,
	       attr_6,
	       attr_7,
	       attr_8,
	       attr_9,
	       attr_10,
	       merch_type,
	       delete_f,
	       banner_id,
	       created_user,
	       created_ts,
	       updated_user,
	       updated_ts
    FROM   staging.csv_merch_level1;
BEGIN
    v_step := 1;
    FOR rec_cur_csv_merch_level1 IN cur_csv_merch_level1 LOOP
        BEGIN
            INSERT INTO staging.stg_merch_level1 
			     (merch_level1_id,
	              merch_level1_name,
	              merch_level1_desc,
	              active_f,
	              retailer_merch_level1_id,
	              attr_1,
	              attr_2,
	              attr_3,
	              attr_4,
	              attr_5,
	              attr_6,
	              attr_7,
	              attr_8,
	              attr_9,
	              attr_10,
	              merch_type,
	              delete_f,
	              banner_id,
	              created_user,
	              created_ts,
	              updated_user,
	              updated_ts)
            VALUES 
			     (rec_cur_csv_merch_level1.merch_level1_id,
	              rec_cur_csv_merch_level1.merch_level1_name,
	              rec_cur_csv_merch_level1.merch_level1_desc,
	              rec_cur_csv_merch_level1.active_f,
	              rec_cur_csv_merch_level1.merch_level1_id,
	              rec_cur_csv_merch_level1.banner_id,
	              rec_cur_csv_merch_level1.attr_2,
	              rec_cur_csv_merch_level1.attr_3,
	              rec_cur_csv_merch_level1.attr_4,
	              rec_cur_csv_merch_level1.attr_5,
	              rec_cur_csv_merch_level1.attr_6,
	              rec_cur_csv_merch_level1.attr_7,
	              rec_cur_csv_merch_level1.attr_8,
	              rec_cur_csv_merch_level1.attr_9,
	              rec_cur_csv_merch_level1.attr_10,
	              rec_cur_csv_merch_level1.merch_type,
	              rec_cur_csv_merch_level1.delete_f,
	              rec_cur_csv_merch_level1.banner_id,
	              rec_cur_csv_merch_level1.created_user,
	              rec_cur_csv_merch_level1.created_ts,
	              rec_cur_csv_merch_level1.updated_user,
	              rec_cur_csv_merch_level1.updated_ts);
	    EXCEPTION 
            WHEN OTHERS THEN
                 INSERT INTO staging.error_logs(error_id, error_function_name, error_step_number, error_state, error_msg, primarykeyid) 
                 VALUES(nextval('staging.error_logs_pk_seq'), 'fn_full_load_to_stg_merch_level1', v_step, sqlstate, sqlerrm, rec_cur_csv_merch_level1.merch_level1_id);
				 RETURN 1;
	    END;
    END LOOP;
    v_step := 2;
	RETURN 0;
END;
$function$
;
